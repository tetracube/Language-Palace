CREATE TABLE IF NOT EXISTS "PARTIDA" (
	"ID"	INTEGER,
	"FECHA"	INTEGER,
	"PELO"	TEXT NOT NULL,
	"OJOS"	TEXT NOT NULL,
	"PIEL"	TEXT NOT NULL,
	"VESTIMENTA"	TEXT NOT NULL,
	"APODO"	TEXT NOT NULL,
	"EDAD"	INTEGER NOT NULL,
	"GENERO"	TEXT NOT NULL,
	"L1"	TEXT NOT NULL,
	"L2"	TEXT NOT NULL,
	"NACIONALIDAD"	TEXT NOT NULL,
	"PAIS"	TEXT NOT NULL,
	PRIMARY KEY("FECHA","ID")
);
CREATE TABLE IF NOT EXISTS "NIVEL" (
	"ID"	INTEGER,
	"FECHA_PARTIDA"	INTEGER,
	"PARTIDA"	INTEGER,
	"NUMERO"	INTEGER NOT NULL,
	"TIEMPO_COMPLETAR"	INTEGER NOT NULL,
	"PALABRAS"	TEXT NOT NULL,
	PRIMARY KEY("ID","FECHA_PARTIDA","PARTIDA"),
	FOREIGN KEY("FECHA_PARTIDA") REFERENCES "PARTIDA"("FECHA"),
	FOREIGN KEY("PARTIDA") REFERENCES "PARTIDA"("ID")
);
CREATE TABLE IF NOT EXISTS "PRUEBA" (
	"ID"	INTEGER,
	"FECHA_PARTIDA"	INTEGER,
	"PARTIDA"	INTEGER,
	"NIVEL"	INTEGER,
	"TIPO"	TEXT NOT NULL,
	"PUNTUACION"	REAL NOT NULL,
	"FALLOS"	TEXT NOT NULL,
	"CANTIDAD_FALLOS"	INTEGER NOT NULL,
	"CANTIDAD_ACIERTOS"	INTEGER NOT NULL,
	"PALABRAS"	TEXT NOT NULL,
	FOREIGN KEY("NIVEL") REFERENCES "NIVEL"("ID"),
	PRIMARY KEY("ID","FECHA_PARTIDA","PARTIDA","NIVEL")
);
